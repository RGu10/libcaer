548 static inline bool caerEventPacketEquals(caerEventPacketHeaderConst firstPacket,
549  caerEventPacketHeaderConst secondPacket) {
550  // If any of the packets is NULL, they can't be equal.
551  if (firstPacket == NULL || secondPacket == NULL) {
552  return (false);
553  }
554 
555  // If both packets are the same packet (pointer equal),
556  // they are of course indeed equal packets.
557  if (firstPacket == secondPacket) {
558  return (true);
559  }
560 
561  // Actually compare memory now. We compare header equality, and
562  // all events up to eventNumber. The remaining events up to
563  // eventCapacity are by definition all zeroed out, so must be
564  // equal, if the capacity is the same, which it is, as we check
565  // for that when ensuring header equality.
566  if (memcmp(firstPacket, secondPacket, CAER_EVENT_PACKET_HEADER_SIZE) != 0) {
567  return (false);
568  }
569 
570  size_t memCmpSize = (size_t) (caerEventPacketHeaderGetEventNumber(firstPacket)
571  * caerEventPacketHeaderGetEventSize(firstPacket));
572  if (memcmp(((const uint8_t *) firstPacket) + CAER_EVENT_PACKET_HEADER_SIZE,
573  ((const uint8_t *) secondPacket) + CAER_EVENT_PACKET_HEADER_SIZE, memCmpSize) != 0) {
574  return (false);
575  }
576 
577  return (true);
578 }

610 static inline void caerEventPacketClean(caerEventPacketHeader packet) {
611  // Handle empty event packets.
612  if (packet == NULL) {
613  return;
614  }
615 
616  // Calculate needed memory for new event packet.
617  int32_t eventValid = caerEventPacketHeaderGetEventValid(packet);
618  int32_t eventNumber = caerEventPacketHeaderGetEventNumber(packet);
619 
620  // If we have no invalid events, we're already done.
621  if (eventValid == eventNumber) {
622  return;
623  }
624 
625  int32_t eventSize = caerEventPacketHeaderGetEventSize(packet);
626  int32_t eventCapacity = caerEventPacketHeaderGetEventCapacity(packet);
627 
628  // Move all valid events close together. Must check every event for validity!
629  size_t offset = CAER_EVENT_PACKET_HEADER_SIZE;
630 
631  CAER_ITERATOR_VALID_START(packet, const void *)
632  void *dest = ((uint8_t *) packet) + offset;
633 
634  if (dest != caerIteratorElement) {
635  memcpy(dest, caerIteratorElement, (size_t) eventSize);
636  offset += (size_t) eventSize;
637  }
638  }
639 
640  // Reset remaining memory, up to capacity, to zero (all events invalid).
641  memset(((uint8_t *) packet) + offset, 0, (size_t) ((eventCapacity - eventValid) * eventSize));
642 
643  // Event capacity remains unchanged, event number shrunk to event valid number.
644  caerEventPacketHeaderSetEventNumber(packet, eventValid);
645 }

665 static inline caerEventPacketHeader caerEventPacketResize(caerEventPacketHeader packet, int32_t newEventCapacity) {
666  if (packet == NULL || newEventCapacity <= 0) {
667  return (NULL);
668  }
669 
670  // Always clean for consistency with shrink case (side-effects guarantee).
671  caerEventPacketClean(packet);
672 
673  int32_t oldEventCapacity = caerEventPacketHeaderGetEventCapacity(packet);
674 
675  if (oldEventCapacity == newEventCapacity) {
676  // Nothing to do in this case.
677  return (packet);
678  }
679 
680  int32_t eventSize = caerEventPacketHeaderGetEventSize(packet);
681  size_t newEventPacketSize = CAER_EVENT_PACKET_HEADER_SIZE + (size_t) (newEventCapacity * eventSize);
682 
683  // Reallocate memory used to hold events.
684  packet = (caerEventPacketHeader) realloc(packet, newEventPacketSize);
685  if (packet == NULL) {
686  caerLog(CAER_LOG_CRITICAL, "Event Packet",
687  "Failed to reallocate %zu bytes of memory for resizing Event Packet of capacity %"
688  PRIi32 " to new capacity of %" PRIi32 ". Error: %d.", newEventPacketSize, oldEventCapacity,
689  newEventCapacity, errno);
690  return (NULL);
691  }
692 
693  if (newEventCapacity > oldEventCapacity) {
694  // Capacity increased: we simply zero out the newly added events.
695  size_t oldEventPacketSize = CAER_EVENT_PACKET_HEADER_SIZE + (size_t) (oldEventCapacity * eventSize);
696 
697  memset(((uint8_t *) packet) + oldEventPacketSize, 0,
698  (size_t) ((newEventCapacity - oldEventCapacity) * eventSize));
699  }
700  else {
701  // Capacity decreased: the events were cleaned, so eventValid == eventNumber.
702  // They also are all together in memory. Thus we can simply keep the current
703  // eventValid/eventNumber counts if the capacity is still bigger or equal to
704  // them, or, if new capacity is smaller, we reset them to that value.
705  int32_t oldEventNumber = caerEventPacketHeaderGetEventNumber(packet);
706 
707  if (newEventCapacity < oldEventNumber) {
708  caerEventPacketHeaderSetEventValid(packet, newEventCapacity);
709  caerEventPacketHeaderSetEventNumber(packet, newEventCapacity);
710  }
711  }
712 
713  // Update capacity header field.
714  caerEventPacketHeaderSetEventCapacity(packet, newEventCapacity);
715 
716  return (packet);
717 }
