243 static inline void caerConfigurationEventValidate(caerConfigurationEvent event, caerConfigurationEventPacket packet) {
244  if (!caerConfigurationEventIsValid(event)) {
245  SET_NUMBITS8(event->moduleAddress, VALID_MARK_SHIFT, VALID_MARK_MASK, 1);
246 
247  // Also increase number of events and valid events.
