234 static inline void caerPolarityEventValidate(caerPolarityEvent event, caerPolarityEventPacket packet) {
235  if (!caerPolarityEventIsValid(event)) {
236  SET_NUMBITS32(event->data, VALID_MARK_SHIFT, VALID_MARK_MASK, 1);

238  // Also increase number of events and valid events.
