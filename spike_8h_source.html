233 static inline void caerSpikeEventValidate(caerSpikeEvent event, caerSpikeEventPacket packet) {
234  if (!caerSpikeEventIsValid(event)) {
235  SET_NUMBITS32(event->data, VALID_MARK_SHIFT, VALID_MARK_MASK, 1);
236 
237  // Also increase number of events and valid events.
238  // Only call this on (still) invalid events!
239  caerEventPacketHeaderSetEventNumber(&packet->packetHeader,
240  caerEventPacketHeaderGetEventNumber(&packet->packetHeader) + 1);
241  caerEventPacketHeaderSetEventValid(&packet->packetHeader,
242  caerEventPacketHeaderGetEventValid(&packet->packetHeader) + 1);
243  }
244  else {
245  caerLog(CAER_LOG_CRITICAL, "Spike Event", "Called caerSpikeEventValidate() on already valid event.");
246  }
247 }

258 static inline void caerSpikeEventInvalidate(caerSpikeEvent event, caerSpikeEventPacket packet) {
